# ====================
# DATABASE SETUP
# ====================
#
# 1. Install MySQL 8.0
# 2. Create database:
#    mysql -u root -p
#    CREATE DATABASE suvidha;
#    EXIT;
#
# 3. Run migration:
#    cd Suvidha/backend
#    node scripts/water-migrate.js
#
# 4. Seed data:
#    node scripts/water-seed.js
#
# ====================

# ====================
# ENVIRONMENT VARIABLES
# ====================
#
# Create .env file in Suvidha/backend/:
#
# PORT=5000
# NODE_ENV=development
# DB_HOST=localhost
# DB_USER=root
# DB_PASSWORD=your_mysql_password
# DB_NAME=suvidha
# DB_PORT=3306
# JWT_SECRET=your_secret_key
# JWT_EXPIRE=24h
#
